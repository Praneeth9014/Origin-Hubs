import java.util.Arrays;
import java.util.Random;
import java.util.Scanner;

public class BinarySearchObjects {

   private static final String ALPHA_NUMERIC_STRING = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

   public static void main(String[] args) {

       Employee1[] employees = new Employee1[10];
       Scanner scan = new Scanner(System.in);
       Random r = new Random();
       for (int i = 0; i < employees.length; i++) {

           employees[i] = new Employee1(randomString(10), randomString(10), r.nextInt(20000));
       }

       Arrays.sort(employees, Employee1.FruitNameComparator);
       for (Employee1 employee1 : employees) {

           System.out.println(employee1.toString());
       }

       System.out.print("Enter the first name and last name of the employee: ");
       String name = scan.nextLine();

       int result = binarySearch(employees, name);

       if (result == -1)
           System.out.println("Element not present");
       else
           System.out.println("Element found at " + "index " + result);
   }

   public static String randomString(int count) {
       StringBuilder builder = new StringBuilder();
       while (count-- != 0) {
           int character = (int) (Math.random() * ALPHA_NUMERIC_STRING.length());
           builder.append(ALPHA_NUMERIC_STRING.charAt(character));
       }
       return builder.toString();
   }

   public static int binarySearch(Employee1[] employees, String name) {
       int low = 0;
       int high = employees.length - 1;
       while (low <= high) {
           int mid = (low + high) / 2;
           int res = name.compareTo((employees[mid].getFirstName() + " " + employees[mid].getLastName()));
          
           if ( res== 0)
               return mid;
           if (res > 0)
               low = mid + 1;
           else
               high = mid - 1;
       }

       return -1;
   }
}
